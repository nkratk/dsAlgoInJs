Priority Queue

  • FIFO - Not First in first out Ex: Police station or call center
  • Enqueue - priority, Dequeue
		class PriorityQueue {
			constructor () {
				this.list = [];
			}
			enQueue (val, priority) {
				if(priority) {
					//add to first
					this.list.unshift(val);
				} else {
					//add to last
					this.list.push(val);
				}
				return this.list;
				
			}
			deQueue () {
				//remove from first
				this.list.shift();
				return this.list;
			}
			peek () {
				//head - get the first one
				return this.list[0];
			}
		}
